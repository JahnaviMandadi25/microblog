pipeline {
    agent any

    stages {
        stage('SCM') {
            steps {
                // Checking out the source code from SCM
                checkout scm
            }
        }

        stage('Build') {
            steps {
                sh 'echo Building the application...'
                // Add your build commands here
            }
        }

        stage('SonarQube Analysis') {
            steps {
                script {
                    // Trigger SonarQube analysis
                    def scannerHome = tool 'SonarScanner'
                    withSonarQubeEnv('SonarQube') {
                        sh "${scannerHome}/bin/sonar-scanner"
                    }
                }
            }
        }

        stage('Deploy') {
            steps {
                sh 'echo Deploying the application...'
                // Add your deployment commands here, e.g., Docker, file transfer, etc.
            }
        }

        stage('Integration Tests') {
            steps {
                sh 'echo Running integration tests...'
                // Add your Selenium or other integration test commands here
            }
        }
    }

    post {
        always {
            echo 'Cleaning up workspace...'
            cleanWs()
        }
        failure {
            mail to: 'mandadij@uwindsor.ca',
                 subject: "Build failed in Jenkins: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                 body: "Something is wrong in ${env.JOB_NAME} #${env.BUILD_NUMBER}\nPlease check Jenkins for details."
        }
    }
}
