pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                sh 'echo Building the application...'

            }
        }

        stage('SonarQube Analysis') {
            steps {
                script {
                    // This is where the SonarQube analysis is triggered
                    def scannerHome = tool 'SonarQube Scanner';
                    withSonarQubeEnv('SonarQube') {
                        sh "${scannerHome}/bin/sonar-scanner"
                    }
                }
            }
        }

        // More stages like deploy and tests can go here
    }

    post {
        always {
            echo 'Cleaning up workspace...'
            cleanWs()
        }
        failure {
            mail to: 'mandadij@uwindsor.ca',
                 subject: "Build failed in Jenkins: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                 body: "Something is wrong in ${env.JOB_NAME} #${env.BUILD_NUMBER}\nPlease check Jenkins for details."
        }
    }
}
